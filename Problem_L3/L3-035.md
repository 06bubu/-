**提示 2：** 分别考虑奇数偶数对应什么情况考虑，怎样去转移

首先我们定 $dp[u][j]$ 定义为：
对于u的子树，当 $j$ 为0时，子树中 $0$ 比 $1$ 多一个
当 $j$ 为1时，子树中 $0$ 和 $1$ 数量相等,
当 $j$ 为2时，子树中 $0$ 比 $1$ 少一个。

我们可以注意到对于节点为偶数的树来说，我们要想让他成为一个美丽的树，那么我们只能 黑白节点数数量相等，也就是只有 $dp[u][1]$ 有意义。
同样对于数量为奇数节点的树，我们要想让他成为一个美丽的树，那么我们只能 黑白节点数的数量差一个，也就是只有 $dp[u][0]$ 和 $dp[u][2]$ 有意义。

然后对于u来看，记他的子节点为v，如果v的节点数为偶数，直接 $dp[v][1]$加上即可
如果v的节点数为奇数，我们可以把所有这样的都存储起来，我们在这里假定一共有 $b$ 个这样的，然后问题就转化到，从这一堆数据里边选择 $a$ 个 $dp[v][0]$ ,$b-a$ 个 $dp[v][2]$
 

时间复杂度为 $\mathcal{O}(nlogn)$ 。

### 具体代码如下

C++ 做法如下

```cpp []
#include <bits/stdc++.h>
using namespace std;
#define int long long
const int INF = 1e18;

int n;

void solve() {
    cin>>n;
    vector<vector<int>> e(n+1);
    vector<int> col(n+1);//颜色colour
    vector<int> a(n+1);//代价
    for(int i=1;i<=n;i++) {
        cin>>col[i]>>a[i];
        int len;cin>>len;
        for(int j=1,x;j<=len;j++) {
            cin >> x;
            e[i].push_back(x);
        }
    }
    vector<vector<int>> dp(n+1,vector<int>(3,INF));
    //0 : 0>1  1:0=1  2 : 0<1
    auto dfs = [&](auto&& dfs,int u) -> int {
        if(col[u] == 0) dp[u][0] = 0,dp[u][2] = a[u];
        else dp[u][0]=a[u],dp[u][2] = 0;

        int num = 1;
        vector<array<int,2>> tmp;
        for(int v : e[u]) {
            int cnt = dfs(dfs,v);
            if(cnt % 2 == 0) {
                dp[u][0] += dp[v][1];
                dp[u][2] += dp[v][1];
            } else {
                tmp.push_back({dp[v][0],dp[v][2]});
            }
            num += cnt;
        }
        tmp.push_back({dp[u][0],dp[u][2]});

        int sum = 0;
        int len = tmp.size();
        for(auto [it1,it2] : tmp) sum += it1;
        sort(tmp.begin(),tmp.end(),[&](array<int,2>& p,array<int,2>& q) -> bool {
            int num1 = p[1] - p[0];
            int num2 = q[1] - q[0];
            return num1 < num2;
        });
        for(int i=0;i<len/2;i++) {
            sum -= tmp[i][0];
            sum += tmp[i][1];
        }
        if((((int)tmp.size())%2) == 0) {//选一半
            //往上传递dp[u][1]
            //然后把一半变成2  
            dp[u][1] = sum;
        } else {
            //往上传递dp[u][0] and dp[u][2]
            dp[u][0] = sum;
            sum -= tmp[len/2][0];
            sum += tmp[len/2][1];
            dp[u][2] = sum;
        }
        return num;
    };

    int num = dfs(dfs,1);
    if(num%2 == 0) cout<<dp[1][1]<<endl;
    else cout<<min(dp[1][0],dp[1][2])<<endl;
}

signed main()
{
    ios_base::sync_with_stdio(false);
    std::cin.tie(0);
    std::cout.tie(0);
    int _ = 1;
    // std::cin >> _;
    while (_--)
        solve();
    return 0;
}
```
